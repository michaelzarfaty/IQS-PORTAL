  { id: 1, name: "TLS Application Fee", type: "Main Fee", amount: 1500.00, description: "Standard TLS processing fee" },
  { id: 2, name: "SOS Fee", type: "State Fee", amount: 0.00, description: "Secretary of State Fee (Varies by State)" },
];

const Invoicing = () => {
  const { role } = useRole();

  const [fees, setFees] = useState<any[]>([]);
  const [isLoading, setIsLoading] = useState(true);

  useEffect(() => {
    fetchFees();
  }, []);

  const fetchFees = async () => {
    try {
      const { data, error } = await supabase.from('pricing_items').select('*').order('created_at', { ascending: true });
      if (error) throw error;
      
      if (data && data.length > 0) {
        setFees(data);
      } else {
        setFees(defaultFees); // Fallback if empty
      }
    } catch (error) {
      console.error("Error fetching fees:", error);
      // Fallback to defaults or local storage if offline/no table
      setFees(defaultFees);
    } finally {
      setIsLoading(false);
    }
  };

  const handleDeleteFee = async (id: number | string) => {
    try {
      // Optimistic update
      setFees(fees.filter(f => f.id !== id));
      
      // If it's a default fee (no uuid), just remove from state
      if (typeof id === 'number') return;
      
      const { error } = await supabase.from('pricing_items').delete().eq('id', id);
      if (error) throw error;
      toast.success("Fee deleted successfully");
    } catch (error) {
      console.error("Error deleting fee:", error);
      toast.error("Failed to delete fee");
      fetchFees(); // Revert on failure
    }
  };

  const handleClearAll = async () => {
    try {
      setFees([]);
      const { error } = await supabase.from('pricing_items').delete().neq('id', '00000000-0000-0000-0000-000000000000'); // Delete all
      if (error) throw error;
      toast.success("All fees cleared");

      console.error("Error clearing fees:", error);
      fetchFees();

    const file = e.target.files?.[0];
    if (!file) return;
    processFile(file);
  };

  const handleDrop = (e: React.DragEvent) => {
    e.preventDefault();
    const file = e.dataTransfer.files?.[0];
    if (!file) return;
    processFile(file);
  };

    const reader = new FileReader();
    reader.onload = (event) => {
      const text = event.target?.result as string;
      setCsvData(text);
      toast.success(`File "${file.name}" loaded. Click Import Data to process.`);
    };

          name: cols[0] || "Unknown Fee",
          type: cols[1] || "Custom Fee",
          amount: parseFloat(cols[2]?.replace(/[^0-9.-]+/g, "")) || 0,
          description: cols[3] || ""
        };
      });
      
      // Insert to Supabase
      const { data, error } = await supabase.from('pricing_items').insert(newFees).select();
      
      if (error) {
        // If table doesn't exist yet, just update local state
        console.error("Supabase error (table might not exist):", error);
        const localFees = newFees.map((f, i) => ({ ...f, id: Date.now() + i }));

      } else if (data) {
        setFees([...fees, ...data]);
      }

      toast.error("Failed to parse data. Please check the format.");
