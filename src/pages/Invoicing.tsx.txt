import { useState } from "react";
import { Layout } from "@/components/Layout";
import { Card, CardContent, CardDescription, CardHeader, CardTitle } from "@/components/ui/card";
import { Table, TableBody, TableCell, TableHead, TableHeader, TableRow } from "@/components/ui/table";
import { Button } from "@/components/ui/button";
import { Plus, FileText, Download, Send, CreditCard, Upload } from "lucide-react";
import { Badge } from "@/components/ui/badge";
import { Dialog, DialogContent, DialogDescription, DialogHeader, DialogTitle, DialogTrigger } from "@/components/ui/dialog";
import { Textarea } from "@/components/ui/textarea";
import { toast } from "sonner";
import { Tabs, TabsContent, TabsList, TabsTrigger } from "@/components/ui/tabs";

const defaultFees = [
  { id: 1, name: "IQS Application Fee", type: "Main Fee", amount: 1500.00, description: "Standard IQS processing fee" },
  { id: 2, name: "SOS Fee", type: "State Fee", amount: 0.00, description: "Secretary of State Fee (Varies by State)" },
];

const Invoicing = () => {
  const [csvData, setCsvData] = useState("");
  const [fees, setFees] = useState(defaultFees);

  const handleParseCSV = () => {
    if (!csvData.trim()) return;
    
    try {
      const rows = csvData.split('\n').filter(r => r.trim());
      const newFees = rows.map((row, index) => {
        const cols = row.split('\t').length > 1 ? row.split('\t') : row.split(',');
        return {
          id: Date.now() + index,
          name: cols[0] || "Unknown Fee",
          type: cols[1] || "Custom Fee",
          amount: parseFloat(cols[2]?.replace(/[^0-9.-]+/g, "")) || 0,
          description: cols[3] || ""
        };
      });
      
      setFees([...defaultFees, ...newFees]);
      setCsvData("");
      toast.success("Pricing data imported successfully!");
    } catch (e) {
      toast.error("Failed to parse data. Please check the format.");
    }
  };

  const handleDownloadPDF = () => {
    toast.success("Generating PDF invoice...");
    setTimeout(() => window.print(), 1000);
  };

  const handleStripePayment = () => {
    toast.info("Redirecting to Stripe checkout...");
  };

  return (
    <Layout>
      <div className="p-8 max-w-7xl mx-auto space-y-6">
        <div className="flex justify-between items-center">
          <div>
            <h1 className="text-3xl font-bold tracking-tight">3rd Party Invoicing</h1>
            <p className="text-muted-foreground mt-2">Manage pricing, generate invoices, and track payments.</p>
          </div>
          <div className="flex gap-3">
            <Button variant="outline"><FileText className="w-4 h-4 mr-2" /> Templates</Button>
            <Button><Plus className="w-4 h-4 mr-2" /> Create Invoice</Button>
          </div>
        </div>

        <Tabs defaultValue="pricing" className="w-full">
          <TabsList className="mb-4">
            <TabsTrigger value="pricing">Standard Pricing</TabsTrigger>
            <TabsTrigger value="states">State-by-State Fees</TabsTrigger>
            <TabsTrigger value="invoices">Invoices & Payments</TabsTrigger>
          </TabsList>

          <TabsContent value="pricing" className="space-y-6">
            <Card>
              <CardHeader className="flex flex-row items-center justify-between">
                <div>
                  <CardTitle>Standard Pricing Schedule</CardTitle>
                  <CardDescription>All CLP branding has been updated to IQS.</CardDescription>
                </div>
                <Dialog>
                  <DialogTrigger asChild>
                    <Button variant="outline"><Upload className="w-4 h-4 mr-2" /> Import CSV/Excel</Button>
                  </DialogTrigger>
                  <DialogContent>
                    <DialogHeader>
                      <DialogTitle>Import Pricing Data</DialogTitle>
                      <DialogDescription>
                        Paste your Excel or CSV data directly into the box below.
                      </DialogDescription>
                    </DialogHeader>
                    <Textarea 
                      placeholder="Paste rows here (Name, Category, Amount, Description)..."
                      className="min-h-[200px]"
                      value={csvData}
                      onChange={(e) => setCsvData(e.target.value)}
                    />
                    <Button onClick={handleParseCSV} className="w-full mt-4">Import Data</Button>
                  </DialogContent>
                </Dialog>
              </CardHeader>
              <CardContent>
                <Table>
                  <TableHeader>
                    <TableRow>
                      <TableHead>Fee Name</TableHead>
                      <TableHead>Category</TableHead>
                      <TableHead>Description</TableHead>
                      <TableHead className="text-right">Amount</TableHead>
                    </TableRow>
                  </TableHeader>
                  <TableBody>
                    {fees.map((fee) => (
                      <TableRow key={fee.id}>
                        <TableCell className="font-medium">{fee.name}</TableCell>
                        <TableCell>
                          <Badge variant={fee.type === 'Main Fee' ? 'default' : 'secondary'}>
                            {fee.type}
                          </Badge>
                        </TableCell>
                        <TableCell className="text-muted-foreground">{fee.description}</TableCell>
                        <TableCell className="text-right font-medium">
                          ${fee.amount.toFixed(2)}
                        </TableCell>
                      </TableRow>
                    ))}
                  </TableBody>
                </Table>
              </CardContent>
            </Card>
          </TabsContent>

          <TabsContent value="states" className="space-y-6">
            <Card>
              <CardHeader>
                <CardTitle>State-by-State Configuration</CardTitle>
                <CardDescription>Configure specific SOS and processing fees per state.</CardDescription>
              </CardHeader>
              <CardContent>
                <Table>
                  <TableHeader>
                    <TableRow>
                      <TableHead>State</TableHead>
                      <TableHead>SOS Fee</TableHead>
                      <TableHead>Processing Time</TableHead>
                      <TableHead>Additional Requirements</TableHead>
                    </TableRow>
                  </TableHeader>
                  <TableBody>
                    <TableRow>
                      <TableCell className="font-medium">Texas (TX)</TableCell>
                      <TableCell>$115.00</TableCell>
                      <TableCell>2-3 Weeks</TableCell>
                      <TableCell className="text-muted-foreground">Fingerprinting required</TableCell>
                    </TableRow>
                    <TableRow>
                      <TableCell className="font-medium">Florida (FL)</TableCell>
                      <TableCell>$250.00</TableCell>
                      <TableCell>4-6 Weeks</TableCell>
                      <TableCell className="text-muted-foreground">Credit report required</TableCell>
                    </TableRow>
                    <TableRow>
                      <TableCell className="font-medium">California (CA)</TableCell>
                      <TableCell>$330.00</TableCell>
                      <TableCell>6-8 Weeks</TableCell>
                      <TableCell className="text-muted-foreground">Bond required</TableCell>
                    </TableRow>
                  </TableBody>
                </Table>
              </CardContent>
            </Card>
          </TabsContent>

          <TabsContent value="invoices" className="space-y-6">
            <Card>
              <CardHeader>
                <CardTitle>Recent Invoices</CardTitle>
                <CardDescription>Latest 3rd party billing</CardDescription>
              </CardHeader>
              <CardContent className="space-y-4">
                <div className="flex items-center justify-between p-4 border rounded-lg">
                  <div>
                    <div className="font-medium text-lg">INV-2024-001</div>
                    <div className="text-sm text-muted-foreground mt-1">Vendor A • $1,500.00</div>
                  </div>
                  <div className="flex items-center gap-3">
                    <Badge variant="outline" className="bg-green-50 text-green-700 border-green-200">Paid</Badge>
                    <Button variant="outline" size="sm" onClick={handleDownloadPDF}>
                      <Download className="w-4 h-4 mr-2" /> PDF
                    </Button>
                  </div>
                </div>
                <div className="flex items-center justify-between p-4 border rounded-lg">
                  <div>
                    <div className="font-medium text-lg">INV-2024-002</div>
                    <div className="text-sm text-muted-foreground mt-1">Vendor B • $1,615.00</div>
                  </div>
                  <div className="flex items-center gap-3">
                    <Badge variant="outline" className="bg-yellow-50 text-yellow-700 border-yellow-200">Pending</Badge>
                    <Button variant="outline" size="sm" onClick={handleDownloadPDF}>
                      <Download className="w-4 h-4 mr-2" /> PDF
                    </Button>
                    <Button size="sm" onClick={handleStripePayment}>
                      <CreditCard className="w-4 h-4 mr-2" /> Pay with Stripe
                    </Button>
                  </div>
                </div>
              </CardContent>
            </Card>
          </TabsContent>
        </Tabs>
      </div>
    </Layout>
  );
};

export default Invoicing;
