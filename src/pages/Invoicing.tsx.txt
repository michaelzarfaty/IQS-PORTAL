  { id: 1, name: "TLS Application Fee", type: "Main Fee", amount: 1500.00, description: "Standard TLS processing fee" },
  { id: 2, name: "SOS Fee", type: "State Fee", amount: 0.00, description: "Secretary of State Fee (Varies by State)" },
];

    const file = e.target.files?.[0];
    if (!file) return;

    const reader = new FileReader();
    reader.onload = (event) => {
      const text = event.target?.result as string;
      setCsvData(text);
    };
    reader.readAsText(file);
  };

    if (!csvData.trim()) return;
    
    try {
      const rows = csvData.split('\n').filter(r => r.trim());
      const newFees = rows.map((row, index) => {
        const cols = row.split('\t').length > 1 ? row.split('\t') : row.split(',');
        return {
          id: Date.now() + index,
          name: cols[0] || "Unknown Fee",
          type: cols[1] || "Custom Fee",
          amount: parseFloat(cols[2]?.replace(/[^0-9.-]+/g, "")) || 0,
          description: cols[3] || ""
        };
      });
      
      setFees([...defaultFees, ...newFees]);
      setCsvData("");
      toast.success("Pricing data imported successfully!");
    } catch (e) {
      toast.error("Failed to parse data. Please check the format.");
    }
  };

    toast.success("Generating PDF invoice...");
    setTimeout(() => window.print(), 1000);

                      </CardHeader>

                    <DialogHeader>
